## 二、数据类型定义

```typescript
/**
 * 水量趋势分析相关类型
 */

export type TimeType = 'day' | 'month' | 'year' | 'total'
export type ChartType = 'curve' | 'volume'

export interface SummaryData {
  dailyVolume: number      // 处理水量 m³
  dailyHours?: number      // 可选，净水厂不需要
  updateTime: string
}

export interface CurveChartData {
  time: string[]
  inflowRate: number[]     // 进水流量 m³/h
  outflowRate: number[]    // 出水流量 m³/h
}

export interface VolumeChartData {
  time: string[]
  volume: number[]         // 处理水量 m³
}

export type ChartData = CurveChartData | VolumeChartData

export interface TableRow {
  stationName: string
  time: string
  dailyVolume: number
  inflowRate: number
  outflowRate: number
  hourlyVolume?: number
  weather: string
  temperature: number
  windSpeed: number
}
```

## 三、API接口

需要创建 `src/api/analysis/waterTrend.ts`：

```typescript
import { request } from '@/utils/request'
import { generateWaterVolumeTrend, generateHistoryRangeData } from '@/utils/waterTreatmentMockData'
import type { TimeType, ChartType, SummaryData, ChartData, TableRow } from '@/api/types/analysis/waterTrend'

/**
 * 获取水量趋势数据
 */
export async function getWaterVolumeTrend(params: {
  stationIds: string[]
  timeType: TimeType
  date: string
  chartType: ChartType
}): Promise<{
  code: number
  message: string
  data: {
    summary: SummaryData
    chartData: ChartData
    tableData: TableRow[]
  }
}> {
  if (import.meta.env.MOCK !== 'false') {
    await new Promise(resolve => setTimeout(resolve, 500))
    
    const date = new Date(params.date)
    const trendData = generateWaterVolumeTrend(
      params.timeType as 'day' | 'month' | 'year',
      date
    )
    
    return {
      code: 200,
      message: 'success',
      data: {
        summary: {
          dailyVolume: trendData.volume.reduce((a, b) => a + b, 0),
          updateTime: new Date().toISOString()
        },
        chartData: params.chartType === 'curve' 
          ? { time: trendData.time, inflowRate: trendData.inflowRate, outflowRate: trendData.outflowRate }
          : { time: trendData.time, volume: trendData.volume },
        tableData: trendData.time.map((t, idx) => ({
          stationName: '鹤问湖水质净化厂',
          time: t,
          dailyVolume: trendData.volume[idx],
          inflowRate: trendData.inflowRate[idx],
          outflowRate: trendData.outflowRate[idx],
          weather: '晴',
          temperature: 20 + Math.random() * 10,
          windSpeed: 5 + Math.random() * 10
        }))
      }
    }
  }
  
  return request.get('/api/analysis/water-trend', { params })
}

/**
 * 导出水量趋势数据
 */
export async function exportWaterTrendData(params: {
  stationIds: string[]
  timeType: TimeType
  date: string
  chartType: ChartType
}): Promise<Blob> {
  if (import.meta.env.MOCK !== 'false') {
    await new Promise(resolve => setTimeout(resolve, 800))
    // 返回模拟的Excel文件
    return new Blob(['模拟Excel数据'], { type: 'application/vnd.ms-excel' })
  }
  
  return request.get('/api/analysis/water-trend/export', {
    params,
    responseType: 'blob'
  })
}
```

## 五、数据对比

| 光伏指标 | 净水厂指标 | 单位 | 说明 |
|---------|-----------|------|------|
| 发电量 | 处理水量 | kWh → m³ | 核心指标 |
| 发电功率 | 进水流量 | kW → m³/h | 实时指标 |
| 瞬时辐照 | 出水流量 | W/m² → m³/h | 实时指标 |
| 等效小时 | - | h | 净水厂不需要 |
| 电站天气 | 天气 | - | 保留 |
| 温度 | 温度 | ℃ | 保留 |
| 风速 | 风速 | km/h | 保留（可选） |

## 六、注意事项

1. **数据范围合理性**
   - 进水流量：500-1200 m³/h（考虑昼夜变化）
   - 出水流量：略小于进水流量（约98%）
   - 日处理水量：15000-20000 m³

2. **昼夜变化规律**
   - 白天（6:00-22:00）流量高
   - 夜间（22:00-6:00）流量低
   - 高峰时段（8:00-10:00, 18:00-20:00）流量增加20%

3. **图表颜色**
   - 进水流量：蓝色系 (#00d4ff)
   - 出水流量：绿色系 (#67c23a)
   - 处理水量：金色系 (#FFD700)

4. **单位统一**
   - 流量：m³/h
   - 水量：m³
   - 能耗：kWh
   - 药剂：kg
