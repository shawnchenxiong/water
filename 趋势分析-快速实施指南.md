```typescript
// 第500行附近 - fetchTrendData 函数
import { getWaterVolumeTrend, getHistoryRangeData } from '@/api/analysis/waterTrend'

async function fetchTrendData() {
  loading.value = true
  chartLoading.value = true
  try {
    let response
    
    if (timeType.value === 'history' && dateRange.value && dateRange.value.length === 2) {
      response = await getHistoryRangeData({
        stationIds: [selectedStationId.value],
        timeType: 'day',
        date: formatDate(dateRange.value[0], 'day'),
        chartType: chartType.value,
        startDate: formatDate(dateRange.value[0], 'day'),
        endDate: formatDate(dateRange.value[1], 'day')
      })
    } else {
      response = await getWaterVolumeTrend({
        stationIds: [selectedStationId.value],
        timeType: timeType.value as TimeType,
        date: formatDate(selectedDate.value, timeType.value as TimeType),
        chartType: chartType.value
      })
    }
    
    if (response.code === 200) {
      summary.value = response.data.summary
      chartData.value = response.data.chartData
      tableData.value = response.data.tableData
    }
    
  } catch (error) {
    console.error('获取数据失败:', error)
    ElMessage.error('获取数据失败')
  } finally {
    loading.value = false
    chartLoading.value = false
  }
}
```

## 常见问题

### Q1: 数据不显示？
A: 检查以下几点：
1. 确保 `MOCK` 环境变量不是 `'false'`
2. 检查浏览器控制台是否有错误
3. 确认API函数导入路径正确

### Q2: 图表显示异常？
A: 检查：
1. chartData 的数据结构是否正确
2. series 配置中的 data 字段是否对应正确
3. yAxis 配置是否匹配 series

### Q3: 表格数据不对？
A: 检查：
1. tableData 的字段名是否与表格列的 prop 匹配
2. 数据格式化函数是否正确
3. 数据类型是否正确（number vs string）
